# binutils工具集使用

binutils工具集包含objdump readelf size addr2line as strings

# 命令总结

ELF文件包含

  • 可重定向文件.0
  • 可执行文件.out
  • core文件
  • 动态/静态库

源码头文件地址： /usr/include/elf.h

# readelf: 查看ELF详请

readelf -h main.o

readelf -S main.o

readelf -s main.o

# objdump: 反汇编ELF

objdump -D main.o

objdump -d main.o

objdump -S main.o

# size：查看section大小

size main.o

# nm： 查看符号表

nm -A main.o

# 装载地址

##　查看全局变量初始化和未初始化区别

#include <stdio.h>

int g_buf[200]; // 初始化和未初始化的区别 g_buf[200] = {1, 2, 3, 4}
int main()
{
    int i;
    for (i = 0; i < 10; i++)
    {
        printf("Hello World!\n");
    }
    return 0;
}
