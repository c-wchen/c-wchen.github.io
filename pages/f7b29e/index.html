# JSON序列化

现在大多数网站都采用前后端分离模式，采用JSON作为数据传输，在后端进行传输过程中需要对对象进行序列化，一般采用Jackson和FastJson序列化工具。

# 常用注解

@JsonProperty 序列化、反序列化时，属性的名称 @JsonIgnoreProperties 序列化、反序列化忽略属性，多个时用“,”隔开 @JsonIgnore 序列化、反序列化忽略属性 @JsonAlias 为反序列化期间要接受的属性定义一个或多个替代名称，可以与@JsonProperty一起使用 @JsonInclude 当属性的值为空（null或者""）时，不进行序列化，可以减少数据传输 @JsonFormat 序列化、反序列化时，格式化时间

# 编程式动态过滤

当在使用Json序列化的过程中需要动态进行过滤掉某些字段，这个时候再使用使用@JsonIgnore不太合适了。以下采用@JsonFilter的方式进行动态的过滤掉某些字段。

@Data
@AllArgsConstructor
@NoArgsConstructor
// 序列化不包含NULL
@JsonInclude(JsonInclude.Include.NON_NULL)
// 设置过滤器
@JsonFilter("myFilter")
public class User {
    private Long userId;
    private String userName;
    private String password;
    private Date createTime;
    private Date updateTime;
    private Integer version;
    private Integer deleted;
}

# JsonView绑定视图

使用JsonView可以指定试图动态进行过滤

import java.io.Serializable;

import com.fasterxml.jackson.annotation.JsonView;

public class User implements Serializable{
	private static final long serialVersionUID = -218391826213575747L;

	// 视图1
	public interface OneView{};
	// 视图2 继承视图1
	public interface TwoView extends OneView{};
	
	private String username;
	private String password;
	private String realName;
	private String sex;

	@JsonView(OneView.class)
	public String getUsername() {
		return username;
	}
	
	public void setUsername(String username) {
		this.username = username;
	}
	
	@JsonView(OneView.class)
	public String getPassword() {
		return password;
	}
	
	public void setPassword(String password) {
		this.password = password;
	}

	@JsonView(TwoView.class)
	public String getRealName() {
		return realName;
	}

	public void setRealName(String realName) {
		this.realName = realName;
	}
	
	@JsonView(TwoView.class)
	public String getSex() {
		return sex;
	}

	public void setSex(String sex) {
		this.sex = sex;
	}
}

# 控制层绑定视图

@RestController
@RequestMapping("/user")
public class UserController {

	@GetMapping("/user1")
	@JsonView(OneView.class)
	public User queryUser1() {
		User user = new User();
		user.setUsername("test1");

		user.setRealName("测试");
