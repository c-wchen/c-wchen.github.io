selenium是一个自动化测试框架， 通过python调用浏览器相关API实现浏览器自动化测试， 主要使用webdriver实现浏览器的相关操作

# webdriver

chrome配置

下载chromedriver驱动，把chromedriver.exe复制到C:\Users\USERNAME\AppData\Local\Google\Chrome\Application， 并把文件路径配置到环境变量中

# 启动浏览器

通过option.add_argument('--headless')无界面启动浏览器，如果没有配置chromedriver环境变量，需要指定chromedriver位置（CHROME_DRIVER_PATH）

from selenium import webdriver
from selenium.webdriver.chrome.options import Options
option = Options()
# 无界面启动
# option.add_argument('--headless')
# browser = webdriver.Chrome(executable_path=CHROME_DRIVER_PATH)
# browser = webdriver.Chrome(options=option)
browser = webdriver.Chrome()

# 控制浏览器

# 访问URL
browser.get('https://baidu.com/')
# 刷新页面
browser.refersh()
# 前进
browser.forward()
# 后退
browser.back()
# 截图
browser.save_screenshot('screenshot.png')
# 网页加载后的网页源代码
browser.page_source

# 元素查找

# 单个元素查找
 find_element_by_class_name
 find_element_by_css_selector
 find_element_by_id
 find_element_by_tag_name
 find_element_by_xpath
# 多个元素查找
 find_elements_by_class_name
 find_elements_by_css_selector
 find_elements_by_id
 find_elements_by_tag_name
 find_elements_by_xpath

以上是常见的元素查找函数， 通过css、xpath选择器进行查找浏览器页面的相关元素

# 等待页面加载

  1. 隐式等待

    browser.implicitly_wait(10)

  2. 显式等待

    from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.common.by import By
# 最长等待时间10s，直到找到该元素
element = WebDriverWait(browser, 10).until(
    EC.presence_of_element_located((By.ID, "myDynamicElement"))
)

# WebElement

  1. click: 点击元素
  2. send_keys: 发送键盘事件
# 向input发送test
input.send_keys('test')
from selenium.webdriver.common.keys import Keys
# 向input发送回车
input.send_keys(Keys.RETURN)
  1. rect： 返回元素的宽高和屏幕上的坐标
# 返回结果
 {'height': 44, 'width': 548, 'x': 298, 'y': 188.40000915527344}
  1. get_attribute、get_property: 获取元素属性
  2. text： 获取文本内容
  3. execute_script: 执行JS脚本
browser.execute_script('window.scrollTo(0, document.body.scrollHeight)')
